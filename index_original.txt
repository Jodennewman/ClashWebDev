    const y = radius * Math.sin(radY);
    return { x, y };
  }
  /**
   * Scale factor to simulate perspective flattening
   * The further from "straight on", the smaller the gradient radius.
   */
  function getScaleFactorFromAngles(angleX, angleY) {
    const maxFlatten = 0.2;  // up to 20% flatten
    const distanceFromCenter = Math.sqrt(angleX * angleX + angleY * angleY);
    // If total angle is 90°, that's the max flatten
    const ratio = Math.min(distanceFromCenter / 90, 1);
    return 1 - ratio * maxFlatten;
  }
  /**
   * Update the <radialGradient> attributes to move the pupil
   * based on the given angles (in degrees).
   * @param {number} angleX - horizontal rotation angle
   * @param {number} angleY - vertical rotation angle
   * @param {number} movementRadius - how far the pupil can move from center
   */
  function updateEyeball(angleX, angleY, movementRadius = 20) {
    // Get the XY offset for the pupil
    const { x, y } = getGradientPositionFromAngles(angleX, angleY, movementRadius);
    // Scale the gradient for perspective
    const scaleFactor = getScaleFactorFromAngles(angleX, angleY);
    const newRadius = baseR * scaleFactor;
    // Apply changes to the radial gradient
    gradient.setAttribute("fx", (baseX + x).toFixed(2));
    gradient.setAttribute("fy", (baseY + y).toFixed(2));
    gradient.setAttribute("r", newRadius.toFixed(2));
  }
  // 4) Return an object so your main code can use these
  return {
    baseX,
    baseY,
    baseR,
    updateEyeball,
    getGradientPositionFromAngles,
    getScaleFactorFromAngles
  };
}
